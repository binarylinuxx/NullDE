cmake_minimum_required(VERSION 3.10)
project(zwm LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
pkg_check_modules(WAYLAND_CLIENT REQUIRED wayland-client)

include_directories(${WAYLAND_CLIENT_INCLUDE_DIRS})
link_directories(${WAYLAND_CLIENT_LIBRARY_DIRS})
add_definitions(${WAYLAND_CLIENT_CFLAGS_OTHER})

# inih (INI parser) - add as subdirectory
add_subdirectory(external/inih)
include_directories(external/inih)

add_executable(zwm src/main.cpp)
target_link_libraries(zwm ${WAYLAND_CLIENT_LIBRARIES} inih)
